#
# We set:
# - GD_INCLUDE_DIRS
# - GD_LIBRARIES
# - GD_FOUND
# 

FIND_PATH(GD_INCLUDE_DIR
    gd.h
  PATHS
    ${GD_DIR}/include
    /usr/include
    /usr/local/include
)

FIND_LIBRARY(GD_LIB
  NAMES
    gd bgd
  PATHS
    $(GD_DIR)/lib
)

IF(WIN32)
  SET(GD_SUPPORT_LIBS "")
ELSE(WIN32)
  SET(GD_SUPPORT_LIBS -lm -ljpeg -lpng12)
ENDIF(WIN32)

IF(GD_LIB AND GD_INCLUDE_DIR)
  SET(GD_FOUND TRUE)
  SET(GD_INCLUDE_DIRS ${GD_INCLUDE_DIR})
  SET(GD_LIBRARIES ${GD_LIB} ${GD_SUPPORT_LIBS})
ELSE(GD_LIB AND GD_INCLUDE_DIR)
  SET(GD_FOUND FALSE)
ENDIF(GD_LIB AND GD_INCLUDE_DIR)
